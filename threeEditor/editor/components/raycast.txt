function getMousePosition(event)
{
	var pointX;
	var pointY;
	pointX = ( event.clientX / player.width ) * 2 - 1;
	pointY = - ( event.clientY / player.height ) * 2 + 1;
	return new THREE.Vector2(pointX,pointY);
}
